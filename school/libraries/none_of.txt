Итератор на начало (begin), итератор на символ, следующий за последним (last), компаратор.
Проверяет, НЕ удовлетворяют ли компаратору элементы, находящиеся с begin до last.
Нет
vector<int> a;
bool comp (int a) {
	return a<0;
};
int main () {
	for (int i=0; i<10; i++) vector.push_back(i);
	cout << none_of(a.begin(), a.end(), comp); //Выведет 1, т.е. true
	a[0]=-10;
	cout << none_of(a.begin(), a.end(), comp); //Выведет 0, т.е. false
	return 0;
};
